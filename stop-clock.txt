<script>
  var msec =0;
  var sec =0;
  var min =0;
  var hrs =0;
  function begin (){
    document.getElementById("start").setAttribute('disabled','disabled')
    watch= setInterval(()=>{
      msec++;
      if (msec==100){
        msec=0;
        sec++;
        if(sec==60){
          sec=0;
          min++;
          document.getElementById("min").innerHTML=min+"m";
          if(min==60){
            min=0;
            hrs++;
            document.getElementById("hrs").innerHTML=hrs+"h";
          }
        }
      }
      document.getElementById("msec").innerHTML=msec+"ms";
      document.getElementById("sec").innerHTML=sec+"s";
      
    },10)
  }
  function stop(){
     document.getElementById("start").removeAttribute('disabled','disabled')
     clearInterval(watch)
  }
  function restart(){
    msec=0;
    sec=0;
    min=0;
    hrs=0;
    document.getElementById("msec").innerHTML=msec;
    document.getElementById("sec").innerHTML=sec;
    document.getElementById("min").innerHTML=min;
    document.getElementById("hrs").innerHTML=hrs;

  }

</script>
